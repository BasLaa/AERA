130  void TCPConnection::tcpBackgroundHandler()
131  {
132 
133  // Wait for the connection to become alive
134  std::this_thread::sleep_for(std::chrono::milliseconds(100));
135 
136  int error_code = 0;
137  while (state_ == RUNNING) {
138  // First send all data from the queue
139  std::unique_ptr<TCPMessage> msg = outgoing_queue_->dequeue();
140  while (msg) {
141  error_code = sendMessage(std::move(msg));
142  if (error_code <= 0) {
143  // Error occured while sending message, break the loop and end the thread.
144  break;
145  }
146  msg = std::move(outgoing_queue_->dequeue());
147  }
148  // Check if new data is on the TCP connection to receive
149  timeval tv{ 0, 100000 };
150  int rc = 0;
151  FD_ZERO(&tcp_client_fd_set_);
152  FD_SET(tcp_client_socket_, &tcp_client_fd_set_);
153  rc = ::select(0, &tcp_client_fd_set_, NULL, NULL, &tv);
154  if (rc == 0) {
155  // No messages on the socket, continue the loop.
156  continue;
157  }
158  if (rc == SOCKET_ERROR) {
159  // Something went wrong when receiving the message, break the handler, end the thread.
160  std::cout << "select() == SOCKET_ERROR error: " << WSAGetLastError() << std::endl;
161  break;
162  }
163  auto in_msg = receiveMessage();
164  if (!in_msg) {
165  // Something went wrong when receiving the message, break the handler, end the thread.
166  break;
167  }
168 
169  // Add it to the queue, let the main thread handle them
170  incoming_queue_->enqueue(std::move(in_msg));
171  }
172  // Clear all entries of the queues before shutting down.
173  incoming_queue_->clear();
174  outgoing_queue_->clear();
175 
176  // Close the socket
177  if (tcp_client_socket_ != INVALID_SOCKET) {
178  int err = shutdown(tcp_client_socket_, SD_SEND);
179  if (err != 0) {
180  std::cout << "ERROR: Shutdown of Client Socket failed with error: " << err << std::endl;
181  }
182  closesocket(tcp_client_socket_);
183  WSACleanup();
184  }
185  tcp_client_socket_ = INVALID_SOCKET;
186  }

188  std::unique_ptr<TCPMessage> TCPConnection::receiveMessage()
189  {
190  // Number of bytes received
191  int received_bytes = 0;
192 
193  // Length of read bytes in total (used to ensure split messages are read correctly)
194  uint64_t len_res = 0;
195 
196  // First read the length of the message to expect (8 byte uint64_t)
197  std::string tcp_msg_len_recv_buf_;
198  // To ensure split message is read correctly
199  while (len_res < msg_length_buf_size_) {
200  received_bytes = recv(tcp_client_socket_, &(tcp_msg_len_recv_buf_[len_res]), msg_length_buf_size_ - len_res, 0);
201  if (received_bytes > 0) {
202  // All good
203  len_res += received_bytes;
204  }
205  else if (received_bytes == 0) {
206  // Client closed the connection
207  len_res = -1;
208  std::cout << "Connection closing..." << std::endl;
209  return NULL;
210  }
211  else {
212  // Error occured during receiving
213  std::cout << "recv failed during recv of data length with error: " << WSAGetLastError() << std::endl;
214  closesocket(tcp_client_socket_);
215  WSACleanup();
216  return NULL;
217  }
218  }
219 
220  // Convert the read bytes to uint64_t. Little Endian! Otherwise invert for loop - not implemented, yet.
221  uint64_t msg_len = 0;
222  for (int i = msg_length_buf_size_ - 1; i >= 0; --i)
223  {
224  msg_len <<= 8;
225  msg_len |= (unsigned char)tcp_msg_len_recv_buf_[i];
226  }
227 
228  // Reset read bytes and total read bytes to 0
229  received_bytes = 0;
230  len_res = 0;
231  // Read as many packages as needed to fill the message buffer. Ensures split messages are received correctly.
232  char* buf = new char[msg_len]();
233  while (len_res < msg_len) {
234  received_bytes = recv(tcp_client_socket_, &buf[len_res], msg_len - len_res, 0);
235  if (received_bytes > 0) {
236  len_res += received_bytes;
237  }
238  else if (received_bytes == 0) {
239  return NULL;
240  }
241  else {
242  std::cout << "recv failed during recv of data message with error: " << WSAGetLastError() << std::endl;
243  closesocket(tcp_client_socket_);
244  WSACleanup();
245  return NULL;
246  }
247  }
248 
249  // Parse the byte-stream into a TCPMessage
250  std::unique_ptr<TCPMessage> msg = std::make_unique<TCPMessage>();
251  if (!msg->ParseFromArray(buf, msg_len)) {
252  std::cout << "ERROR: Parsing Message from String failed" << std::endl;
253  return NULL;
254  }
255 
256  delete buf;
257 
258  return msg;
259  }
