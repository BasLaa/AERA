92 r_exec::View *build_view(Timestamp time, Code* rstdin) { // this is application dependent WRT view->sync.
93 
94  r_exec::View *view = new r_exec::View();
95  const uint32 arity = VIEW_ARITY; // reminder: opcode not included in the arity.
96  uint16 write_index = 0;
97  uint16 extent_index = arity + 1;
98 
99  view->code(VIEW_OPCODE) = Atom::SSet(r_exec::View::ViewOpcode_, arity);
100  view->code(VIEW_SYNC) = Atom::Float(View::SYNC_ONCE); // sync on front.
101  view->code(VIEW_IJT) = Atom::IPointer(extent_index); // iptr to injection time.
102  view->code(VIEW_SLN) = Atom::Float(1.0); // sln.
103  view->code(VIEW_RES) = Atom::Float(1); // res is set to 1 upr of the destination group.
104  view->code(VIEW_HOST) = Atom::RPointer(0); // stdin/stdout is the only reference.
105  view->code(VIEW_ORG) = Atom::Nil(); // org.
106 
107  Utils::SetTimestamp(&view->code(extent_index), time);
108 
109  view->references_[0] = rstdin;
110 
111  return view;
112 }
