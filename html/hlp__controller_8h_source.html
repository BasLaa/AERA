141  template<class E> void _store_evidence(Cache<E> *cache, _Fact *evidence) {
142 
143  E e(evidence);
144  cache->CS.enter();
145  auto now = Now();
146  r_code::list<E>::const_iterator _e;
147  for (_e = cache->evidences.begin(); _e != cache->evidences.end();) {
148 
149  if ((*_e).evidence_ == e.evidence_) {
150  // Already stored.
151  cache->CS.leave();
152  return;
153  }
154  if ((*_e).is_too_old(now)) // garbage collection.
155  _e = cache->evidences.erase(_e);
156  else
157  ++_e;
158  }
159  cache->evidences.push_front(e);
160  cache->CS.leave();
161  }
